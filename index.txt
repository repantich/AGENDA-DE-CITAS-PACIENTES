    <script>
        // Estado global
        const state = {
            respuestas: {
                p1: null,
                p2: null,
                p2_detalle: null,
                p3: null,
                derivado: null,
                nombreColega: null
            }
        };

        // PREGUNTA 1
        function responderP1(respuesta, btn) {
            console.log('P1 clicked:', respuesta);
            state.respuestas.p1 = respuesta;
            
            document.querySelectorAll('#pregunta1 .btn-option').forEach(b => b.classList.remove('selected'));
            btn.classList.add('selected');
            
            setTimeout(() => {
                document.getElementById('pregunta1').classList.add('hidden');
                document.getElementById('pregunta2').classList.remove('hidden');
                document.getElementById('pregunta2').scrollIntoView({behavior: 'smooth'});
            }, 300);
        }

        // PREGUNTA 2
        function responderP2(respuesta, btn) {
            console.log('P2 clicked:', respuesta);
            state.respuestas.p2 = respuesta;
            
            document.querySelectorAll('#pregunta2 .btn-option').forEach(b => b.classList.remove('selected'));
            btn.classList.add('selected');
            
            const p2Extra = document.getElementById('p2Extra');
            const p2Label = document.getElementById('p2Label');
            const p2Input = document.getElementById('p2Input');
            const btnContinuar = document.getElementById('btnContinuarP2');
            
            if (respuesta === 'INTERIOR_PROVINCIA') {
                p2Label.textContent = '¿De qué localidad?';
                p2Input.placeholder = 'Ej: Goya, Paso de los Libres, etc.';
                p2Extra.classList.remove('hidden');
                btnContinuar.classList.add('hidden');
            } else if (respuesta === 'OTRA_PROVINCIA_PAIS') {
                p2Label.textContent = '¿De dónde?';
                p2Input.placeholder = 'Ej: Buenos Aires, Paraguay, etc.';
                p2Extra.classList.remove('hidden');
                btnContinuar.classList.add('hidden');
            } else {
                p2Extra.classList.add('hidden');
                btnContinuar.classList.remove('hidden');
            }
            
            if (respuesta !== 'CORRIENTES_CAPITAL') {
                p2Input.oninput = function() {
                    btnContinuar.classList.toggle('hidden', this.value.trim().length < 3);
                };
            }
        }

        function avanzarP3() {
            const p2Input = document.getElementById('p2Input');
            if (!document.getElementById('p2Extra').classList.contains('hidden')) {
                state.respuestas.p2_detalle = p2Input.value.trim();
                if (!state.respuestas.p2_detalle) {
                    alert('⚠️ Por favor complete el campo');
                    return;
                }
            }
            
            document.getElementById('pregunta2').classList.add('hidden');
            document.getElementById('pregunta3').classList.remove('hidden');
            document.getElementById('pregunta3').scrollIntoView({behavior: 'smooth'});
        }

        // PREGUNTA 3
        function responderP3(respuesta, btn) {
            console.log('P3 clicked:', respuesta);
            state.respuestas.p3 = respuesta;
            
            document.querySelectorAll('#pregunta3 > .btn-group .btn-option').forEach(b => b.classList.remove('selected'));
            btn.classList.add('selected');
            
            const p3Sub = document.getElementById('p3Sub');
            const btnVerTurnos = document.getElementById('btnVerTurnos');
            
            if (respuesta === 'CONSULTA_SIMPLE') {
                p3Sub.classList.remove('hidden');
                btnVerTurnos.classList.add('hidden');
            } else {
                p3Sub.classList.add('hidden');
                state.respuestas.derivado = false;
                btnVerTurnos.classList.remove('hidden');
            }
        }

        function responderDerivacion(derivado, btn) {
            state.respuestas.derivado = derivado;
            
            document.querySelectorAll('#p3Sub .btn-group .btn-option').forEach(b => b.classList.remove('selected'));
            btn.classList.add('selected');
            
            const derivacionExtra = document.getElementById('derivacionExtra');
            const btnVerTurnos = document.getElementById('btnVerTurnos');
            const nombreColegaInput = document.getElementById('nombreColega');
            
            if (derivado) {
                derivacionExtra.classList.remove('hidden');
                btnVerTurnos.classList.add('hidden');
                nombreColegaInput.oninput = function() {
                    btnVerTurnos.classList.toggle('hidden', this.value.trim().length < 3);
                };
            } else {
                derivacionExtra.classList.add('hidden');
                btnVerTurnos.classList.remove('hidden');
            }
        }

        function mostrarCalendario() {
            if (!state.respuestas.p3) {
                alert('⚠️ Por favor selecciona el motivo de la consulta');
                return;
            }
            
            if (state.respuestas.p3 === 'CONSULTA_SIMPLE' && state.respuestas.derivado === null) {
                alert('⚠️ Por favor indica si viene derivado');
                return;
            }
            
            if (state.respuestas.derivado) {
                const nombreColega = document.getElementById('nombreColega').value.trim();
                if (!nombreColega) {
                    alert('⚠️ Por favor ingresa el nombre del médico derivante');
                    return;
                }
                state.respuestas.nombreColega = nombreColega;
            }
            
            document.getElementById('pregunta3').classList.add('hidden');
            document.getElementById('calendarioSection').classList.remove('hidden');
            document.getElementById('calendarioSection').scrollIntoView({behavior: 'smooth'});
        }
        
        console.log('Script cargado correctamente');
    </script>
